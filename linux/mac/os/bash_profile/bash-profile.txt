echo "start configuring bash profile"

# 1. Locale & Warnungen
export LANG=en_US.UTF-8
export LC_ALL=en_US.UTF-8

# 2. Homebrew
eval "$(/opt/homebrew/bin/brew shellenv)"

# 3. NVM laden
export NVM_DIR=~/.nvm
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"

# 4. Load Angular CLI autocompletion.
source <(ng completion script)

# 5. Java & Maven Pfad
export PATH="/Users/hek6q32/Han/applications/maven/apache-maven-3.9.9/bin:$PATH"
export JAVA_HOME=/opt/homebrew/Cellar/openjdk@21/21.0.9/libexec/openjdk.jdk/Contents/Home
export PATH="/opt/homebrew/opt/openjdk@21/bin:$PATH"

# 6. Aliase und Funktionen
alias git-fetch='watch -n 1 -x git fetch --all --prune'
alias git-pull='watch -n 1 -x git pull'
alias mvn-build='mvn -DskipTests clean package'

git-push-u(){
  watch -n 1 git push -u "$1" "$2"
}

git-push(){
  watch -n 1 git push
}

echo "bash profile configured"
